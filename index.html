        <table id="leaderboard">

            <thead>

                <tr>

                    <th>Rank</th>

                    <th>Contender</th>

                    <th>Wins</th>

                </tr>

            </thead>

            <tbody>

                <!-- Leaderboard will be populated here -->

            </tbody>

    <script>

        const gameResults = [

            { date: '2024-05-01', contender: 'Alice', score: 90 },

            { date: '2024-05-01', contender: 'Bob', score: 85 },

            { date: '2024-05-01', contender: 'Charlie', score: 80 },

            { date: '2024-05-02', contender: 'Alice', score: 88 },

            { date: '2024-05-02', contender: 'Charlie', score: 92 },

            { date: '2024-05-02', contender: 'Bob', score: 78 },

            { date: '2024-05-03', contender: 'Charlie', score: 89 },

            { date: '2024-05-03', contender: 'Bob', score: 91 },

            { date: '2024-05-03', contender: 'Alice', score: 85 }

        ];


        // Populate game results

        const resultsTableBody = document.getElementById('gameResults').getElementsByTagName('tbody')[0];

        gameResults.forEach(result => {

            const row = resultsTableBody.insertRow();

            row.insertCell(0).innerText = result.date;

            row.insertCell(1).innerText = result.contender;

            row.insertCell(2).innerText = result.score;

        });


        // Calculate wins

        const wins = {};

        gameResults.forEach(result => {

            if (!wins[result.contender]) {

                wins[result.contender] = 0;

            }

            if (result.score === Math.max(...gameResults.filter(r => r.date === result.date).map(r => r.score))) {

                wins[result.contender] += 1;

            }

        });


        // Sort contenders by wins and get top 3

        const topContenders = Object.entries(wins).sort((a, b) => b[1] - a[1]).slice(0, 3);


        // Populate leaderboard

        const leaderboardTableBody = document.getElementById('leaderboard').getElementsByTagName('tbody')[0];

        topContenders.forEach((contender, index) => {

            const row = leaderboardTableBody.insertRow();

            row.insertCell(0).innerText = index + 1;

            row.insertCell(1).innerText = contender[0];

            row.insertCell(2).innerText = contender[1];

        });

    </script>
